# ============================================================
#   MATEMATIKA DISKRIT — BEDA SETANGKUP (SYMMETRIC DIFFERENCE)
#   Implementasi: A ⊕ B = (A ∪ B) - (A ∩ B)
# ============================================================

def gabungan(A: set, B: set) -> set:
    """Menghitung GABUNGAN (Union): A ∪ B"""
    return A | B           # operator | = union bawaan Python

def irisan(A: set, B: set) -> set:
    """Menghitung IRISAN (Intersection): A ∩ B"""
    return A & B           # operator & = intersection bawaan Python

def beda_setangkup(A: set, B: set) -> set:
    """
    Menghitung BEDA SETANGKUP: A ⊕ B = (A ∪ B) − (A ∩ B)
    Elemen yang ada di A atau B, tapi TIDAK di keduanya.
    """
    union        = gabungan(A, B)        # Step 1: A ∪ B
    intersection = irisan(A, B)          # Step 2: A ∩ B
    hasil        = union - intersection   # Step 3: buang irisan
    return hasil

def parse_input(teks: str, nama: str) -> set:
    """Validasi & konversi string input ke set bilangan bulat."""
    elemen = teks.strip().split()
    if not elemen:
        raise ValueError(f"Himpunan {nama} tidak boleh kosong!")
    hasil_set = set()
    for item in elemen:
        if not item.lstrip('-').isdigit():
            raise ValueError(f"'{item}' bukan bilangan bulat!")
        hasil_set.add(int(item))
    return hasil_set

def main():
    A = parse_input(input("Himpunan A: "), "A")
    B = parse_input(input("Himpunan B: "), "B")
    hasil = beda_setangkup(A, B)
    print(f"A ∪ B  = {gabungan(A,B)}")
    print(f"A ∩ B  = {irisan(A,B)}")
    print(f"A ⊕ B  = {hasil}")

if __name__ == "__main__":
    main()
